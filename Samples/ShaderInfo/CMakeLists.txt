FILE(GLOB VK_SAMPLE_SOURCE_FILES ${CMAKE_CURRENT_SOURCE_DIR}/*.cpp)
FILE(GLOB VK_SAMPLE_HEADER_FILES ${CMAKE_CURRENT_SOURCE_DIR}/*.h)

ADD_EXECUTABLE(ShaderInfo ${VK_SAMPLE_SOURCE_FILES} ${VK_SAMPLE_HEADER_FILES} )
TARGET_LINK_LIBRARIES(ShaderInfo vk-sample-common )
ADD_DEPENDENCIES(ShaderInfo vk-sample-common)

# Vulkan dependies headers.
TARGET_INCLUDE_DIRECTORIES(ShaderInfo PRIVATE ${CMAKE_CURRENT_SOURCE_DIR} )


INSTALL(TARGETS ShaderInfo DESTINATION bin)
