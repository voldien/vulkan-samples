FILE(GLOB SOURCE_FILES ${CMAKE_CURRENT_SOURCE_DIR}/*.cpp)
FILE(GLOB HEADER_FILES ${CMAKE_CURRENT_SOURCE_DIR}/*.h)

ADD_EXECUTABLE(StartUpWindow ${SOURCE_FILES} ${HEADER_FILES} )
TARGET_LINK_LIBRARIES(StartUpWindow vk-sample-common )
ADD_DEPENDENCIES(StartUpWindow vk-sample-common)

# Vulkan dependies headers.
TARGET_INCLUDE_DIRECTORIES(StartUpWindow PRIVATE ${CMAKE_CURRENT_SOURCE_DIR} )

INSTALL(TARGETS StartUpWindow DESTINATION bin)
