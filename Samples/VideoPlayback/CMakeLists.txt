FILE(GLOB SOURCE_FILES ${CMAKE_CURRENT_SOURCE_DIR}/*.cpp)
FILE(GLOB HEADER_FILES ${CMAKE_CURRENT_SOURCE_DIR}/*.h)
IF(PKG_CONFIG_FOUND)
	MESSAGE(STATUS "Found pkg-config: ${PKG_CONFIG_EXECUTABLE} -- version ${PKG_CONFIG_VERSION_STRING}.")
    PKG_CHECK_MODULES(AVFORMAT QUIET libavformat)
    PKG_CHECK_MODULES(AVCODEC QUIET libavcodec)
    PKG_CHECK_MODULES(AVUTIL QUIET libavutil)
    PKG_CHECK_MODULES(SWSCALE QUIET libswscale)
ELSE()
	MESSAGE(ERROR "Could not find AV Libraries.")
	# TODO add logic for handling if pkg-config not available.
ENDIF()
ADD_EXECUTABLE(VideoPlayback ${SOURCE_FILES} ${HEADER_FILES} )
TARGET_LINK_LIBRARIES(VideoPlayback ${AVUTIL_LIBRARIES} ${AVFORMAT_LIBRARIES} ${AVCODEC_LIBRARIES} ${SWSCALE_LIBRARIES} fragcore-audio-interface vk-sample-common fragcore-aal)
ADD_DEPENDENCIES(VideoPlayback vk-sample-common)

# Vulkan dependies headers.
TARGET_INCLUDE_DIRECTORIES(VideoPlayback PRIVATE ${CMAKE_CURRENT_SOURCE_DIR} )
