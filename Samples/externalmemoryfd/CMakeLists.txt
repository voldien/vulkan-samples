FILE(GLOB SOURCE_FILES ${CMAKE_CURRENT_SOURCE_DIR}/*.cpp)
FILE(GLOB HEADER_FILES ${CMAKE_CURRENT_SOURCE_DIR}/*.h)

ADD_EXECUTABLE(ExternalMemoryFD ${SOURCE_FILES} ${HEADER_FILES})
TARGET_LINK_LIBRARIES(ExternalMemoryFD vk-sample-common)
ADD_DEPENDENCIES(ExternalMemoryFD vk-sample-common)

# Vulkan dependies headers.
TARGET_INCLUDE_DIRECTORIES(ExternalMemoryFD PRIVATE ${CMAKE_CURRENT_SOURCE_DIR})

INSTALL(TARGETS ExternalMemoryFD DESTINATION bin)
